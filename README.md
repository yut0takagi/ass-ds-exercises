```bash
git clone git@github.com:yut0takagi/ass-ds-exercises.git
cd ass-ds-exercises

# 推奨: 仮想環境の作成
python -m venv .venv
source .venv/bin/activate  # Windows: .venv\Scripts\activate

# 依存のインストール
pip install -U pip
pip install -r requirements.txt

# 入力, 出力ディレクトリを作成（必須）
mkdir -p data
```
